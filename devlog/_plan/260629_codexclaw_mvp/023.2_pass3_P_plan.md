## File change map
### NEW `components/pabcd-state/src/goal-gate.ts`
```ts
export interface PreToolUsePayload {
  hook_event_name: "PreToolUse"; session_id: string; cwd: string;
  tool_name: string; tool_input: unknown; turn_id?: string;
  transcript_path?: string | null;
}
const CREATE_GOAL = "create_goal";
const WARNING = "Use create_goal with objective only. Omit token_budget so the goal stays unlimited; put lifecycle status changes on update_goal.";
export function parsePreToolUse(raw): PreToolUsePayload | null { /* JSON.parse + guard hook_event_name/tool_name/cwd; null on corrupt */ }
function hasInvalidCreateGoalInput(v: unknown): boolean { return isRecord(v) && Object.keys(v).some(k => k !== "objective"); }
export function applyGoalBudgetGuard(p: PreToolUsePayload): string {
  if (p.hook_event_name !== "PreToolUse") return "";
  if (p.tool_name !== CREATE_GOAL) return "";
  if (!hasInvalidCreateGoalInput(p.tool_input)) return "";
  return JSON.stringify({ hookSpecificOutput: { hookEventName: "PreToolUse", permissionDecision: "deny", permissionDecisionReason: WARNING, additionalContext: WARNING } }) + "\n";
}
```
